select la.Title, la.Id, s.* 
from LibraryAssets la
inner join Statuses s 
on s.id = la.StatusId
where la.id = 1

select c.*, lc.* , p.LastName, p.FirstName
from checkouts c
inner join LibraryCards lc 
on lc.Id = c.LibraryCardId
inner join Patrons p
on p.LibraryCardId = lc.Id



delete from Checkouts
delete from CheckoutHistories
